# This cell loads the seaborn Titanic dataset, computes joint and conditional distributions
# needed for Question 2 follow-ups, and prints the key proportions.
import pandas as pd
import seaborn as sns

# Load dataset
titanic_df_q = sns.load_dataset('titanic')

# Keep only needed columns and drop missing
use_df = titanic_df_q[['embarked', 'pclass']].dropna()

# Joint distribution (counts)
joint_counts = use_df.groupby(['embarked', 'pclass']).size().unstack(fill_value=0)
print(joint_counts.head())

# Conditional P(class | embarked)
cond_class_given_embarked = joint_counts.div(joint_counts.sum(axis=1), axis=0)
print(cond_class_given_embarked.head())

# Conditional P(embarked | class)
cond_embarked_given_class = joint_counts.div(joint_counts.sum(axis=0), axis=1)
print(cond_embarked_given_class.head())

# Specific questions:
# 1) Proportion of 3rd class passengers who embarked at Southampton = P(S | class=3)
prop_S_given_class3 = cond_embarked_given_class.loc['S', 3]
print(prop_S_given_class3)

# 2) Proportion of Southampton passengers who were in 3rd class = P(class=3 | S)
prop_class3_given_S = cond_class_given_embarked.loc['S', 3]
print(prop_class3_given_S)

print('Computed conditional distributions and two requested proportions.')

# Code here

import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt


titanic = sns.load_dataset("titanic")


df = titanic[['embarked', 'pclass']].dropna()


dist = (
    df
    .groupby(['embarked', 'pclass'])
    .size()
    .reset_index(name='count')
    .pivot(index='embarked', columns='pclass', values='count')
    .fillna(0)
)


dist_norm = dist.div(dist.sum(axis=1), axis=0)


colors = sns.color_palette("Set2", n_colors=dist_norm.shape[1])
ax = dist_norm.sort_index().plot(
    kind='bar',
    stacked=True,
    color=colors,
    figsize=(8, 5),
    width=0.8
)

plt.title("Distribution of Passenger Class by Embarkation Port (Normalized)")
plt.xlabel("Embarkation Port")
plt.ylabel("Percentage of Passengers")
plt.legend(title="Pclass", loc="best")
plt.gca().yaxis.set_major_formatter(plt.FuncFormatter(lambda y, _: str(int(y * 100)) + "%"))
plt.tight_layout()
plt.show()
